var fromTimeSelect = document.getElementById("from");
var toTimeSelect = document.getElementById("to");

fromTimeSelect.addEventListener("change", function() {
  var selectedFromTime = fromTimeSelect.value;
  
  // Clear existing options
  toTimeSelect.innerHTML = "";
  
  // Add new options based on the selected From Time
  if (selectedFromTime === "8:00 AM") {
    addOption(toTimeSelect, "9:00 AM");
    addOption(toTimeSelect, "10:00 AM");
    addOption(toTimeSelect, "11:00 AM");
    // Add more options as needed
  } else if (selectedFromTime === "9:00 AM") {
    addOption(toTimeSelect, "10:00 AM");
    addOption(toTimeSelect, "11:00 AM");
    addOption(toTimeSelect, "12:00 PM");
    // Add more options as needed
  }
  
  // Function to add option to select element
  function addOption(selectElement, optionText) {
    var option = document.createElement("option");
    option.text = optionText;
    selectElement.add(option);
  }
});
